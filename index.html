@api.depends('account_type')
def _compute_reconcile(self):
    for account in self:
        account.reconcile = account.account_type in ('asset_receivable', 'liability_payable')

Onchange function


@api.onchange('name')
def _onchange_name(self):
    code, name = self._split_code_name(self.name)
    if code:
        self.name = name
        self.code = code
